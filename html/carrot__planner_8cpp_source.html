72  //we need to take the footprint of the robot into account when we calculate cost to obstacles
73  double CarrotPlanner::footprintCost(double x_i, double y_i, double theta_i){
74  if(!initialized_){
75  ROS_ERROR("The planner has not been initialized, please call initialize() to use the planner");
76  return -1.0;
77  }
78 
79  std::vector<geometry_msgs::Point> footprint = costmap_ros_->getRobotFootprint();
80  //if we have no footprint... do nothing
81  if(footprint.size() < 3)
82  return -1.0;
83 
84  //check if the footprint is legal
85  double footprint_cost = world_model_->footprintCost(x_i, y_i, theta_i, footprint);
86  return footprint_cost;
87  }

90  bool CarrotPlanner::makePlan(const geometry_msgs::PoseStamped& start,
91  const geometry_msgs::PoseStamped& goal, std::vector<geometry_msgs::PoseStamped>& plan){
92 
93  if(!initialized_){
94  ROS_ERROR("The planner has not been initialized, please call initialize() to use the planner");
95  return false;
96  }
97 
98  ROS_DEBUG("Got a start: %.2f, %.2f, and a goal: %.2f, %.2f", start.pose.position.x, start.pose.position.y, goal.pose.position.x, goal.pose.position.y);
99 
100  plan.clear();
101  costmap_ = costmap_ros_->getCostmap();
102 
103  if(goal.header.frame_id != costmap_ros_->getGlobalFrameID()){
104  ROS_ERROR("This planner as configured will only accept goals in the %s frame, but a goal was sent in the %s frame.",
105  costmap_ros_->getGlobalFrameID().c_str(), goal.header.frame_id.c_str());
106  return false;
107  }
108 
109  const double start_yaw = tf2::getYaw(start.pose.orientation);
110  const double goal_yaw = tf2::getYaw(goal.pose.orientation);
111 
112  //we want to step back along the vector created by the robot's position and the goal pose until we find a legal cell
113  double goal_x = goal.pose.position.x;
114  double goal_y = goal.pose.position.y;
115  double start_x = start.pose.position.x;
116  double start_y = start.pose.position.y;
117 
118  double diff_x = goal_x - start_x;
119  double diff_y = goal_y - start_y;
120  double diff_yaw = angles::normalize_angle(goal_yaw-start_yaw);
121 
122  double target_x = goal_x;
123  double target_y = goal_y;
124  double target_yaw = goal_yaw;
125 
126  bool done = false;
127  double scale = 1.0;
128  double dScale = 0.01;
129 
130  while(!done)
131  {
132  if(scale < 0)
133  {
134  target_x = start_x;
135  target_y = start_y;
136  target_yaw = start_yaw;
137  ROS_WARN("The carrot planner could not find a valid plan for this goal");
138  break;
139  }
140  target_x = start_x + scale * diff_x;
141  target_y = start_y + scale * diff_y;
142  target_yaw = angles::normalize_angle(start_yaw + scale * diff_yaw);
143 
144  double footprint_cost = footprintCost(target_x, target_y, target_yaw);
145  if(footprint_cost >= 0)
146  {
147  done = true;
148  }
149  scale -=dScale;
150  }
151 
152  plan.push_back(start);
153  geometry_msgs::PoseStamped new_goal = goal;
154  tf2::Quaternion goal_quat;
155  goal_quat.setRPY(0, 0, target_yaw);
156 
157  new_goal.pose.position.x = target_x;
158  new_goal.pose.position.y = target_y;
159 
160  new_goal.pose.orientation.x = goal_quat.x();
161  new_goal.pose.orientation.y = goal_quat.y();
162  new_goal.pose.orientation.z = goal_quat.z();
163  new_goal.pose.orientation.w = goal_quat.w();
164 
165  plan.push_back(new_goal);
166  return (done);
167  }
